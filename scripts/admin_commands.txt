--- QUICK: FULL MANUAL RESTART ---
# If something breaks after a reboot, run the startup script manually
bash /home/ec2-user/startup.sh

--- 1. FRONTEND (REACT UI) ---
# Fix ownership if a previous sudo build broke permissions
sudo chown -R ec2-user:ec2-user /home/ec2-user/private-hub-frontend

# Rebuild the React frontend
cd /home/ec2-user/private-hub-frontend && npm run build

# Deploy fresh build to Nginx
sudo rm -rf /usr/share/nginx/html/* && sudo cp -r dist/* /usr/share/nginx/html/

--- 2. BACKEND (NODE API) ---
# Stop the Node server safely
sudo pkill node

# Start the Node server in the background (detached)
nohup /usr/bin/node /home/ec2-user/private-hub-api/server.js > /home/ec2-user/private-hub-api/server.log 2>&1 &

# Stream live API logs (Ctrl+C to exit)
tail -f /home/ec2-user/private-hub-api/server.log

--- 3. NGINX ---
# Test config for syntax errors (always run before reload)
sudo nginx -t

# Reload Nginx (applies config changes without downtime)
sudo systemctl reload nginx

# Full restart
sudo systemctl restart nginx

# Check status
sudo systemctl status nginx

# Edit config
sudo nano /etc/nginx/nginx.conf

# View access logs
sudo tail -f /var/log/nginx/access.log

# View error logs
sudo tail -f /var/log/nginx/error.log

--- 4. DOCKER & REGISTRY ---
# Check running containers
docker ps

# View registry logs (watch pushes/pulls in real time)
docker logs -f private_hub

# Restart the registry container
docker restart private_hub

# List repositories stored inside the registry
docker exec -it private_hub ls -l /var/lib/registry/docker/registry/v2/repositories/

# Restart the Docker daemon
sudo systemctl restart docker

# Run the registry container with persistent volume (first-time setup)
docker run -d \
  -p 5000:5000 \
  --restart unless-stopped \
  --name private_hub \
  -v /home/ec2-user/registry-data:/var/lib/registry \
  registry:2

# Clone the repo and install dependencies

  -p 5000:5000 \

  --name private_hub \
